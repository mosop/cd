Overview

An example class for EntryDir.dir and EntryDir.any.

class CrystalDir < Cd::EntryDir
  class SpecDir < Cd::EntryDir
    def files
      glob("**/*_spec.cr")
    end
  end

  dir "spec", SpecDir
  any "shard.yml"

  def shard
    YAML.parse(shard_yml.read)
  end

  def version
    shard["version"].to_s
  end
end
